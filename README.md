Usage examples
--------------

Далее три типичных примера:

```Shell
# Найти простые множители для покрытия модуля 99
$ perl fact.pl 99

pi=3 : yes
pi=5 : yes
pi=7 : yes

# Результат:
A = 99;
P = 3 * 5 * 7 = 105;
where P >= A
q[0] = 3 - 1 = 2;
q[1] = 5 - 1 = (2**2);
q[2] = 7 - 1 = 2 * 3;
```

```Shell
# Найти простые множители для покрытия модуля 99,
# только теперь будем использовать множители 6-ти бит.
# Если не получится покрыть модуль 99 6-ти битными
# множителями, то используем для покрытия 7-ми битные,
# то есть мы имеем 1 дополнительный резервный бит для покрытия 
$ perl fact.pl 99 6

pi=37 : no
pi=41 : yes
pi=43 : yes

# Результат:
A = 99;
P = 41 * 43 = 1763;
where P >= A
q[0] = 41 - 1 = (2**3) * 5;
q[1] = 43 - 1 = 2 * 3 * 7;
```

```Shell
# теперь используем 100 битные модули, они находятся в полуинтервале
# от 2 ** 99  =  633825300114114700748351602688
# до 2 ** 100 = 1267650600228229401496703205376
$ perl fact.pl 1267650600228229401496703205375 13

pi=4099 : yes
pi=4111 : yes
pi=4127 : yes
pi=4129 : yes
pi=4133 : yes
pi=4139 : yes
pi=4153 : yes
pi=4157 : yes
pi=4159 : no
pi=4177 : no
pi=4201 : no
pi=4211 : yes

# Результат:
A = 1267650600228229401496703205375;
P = 4099 * 4111 * 4127 * 4129 * 4133 * 4139 * 4153 * 4157 * 4211 = 357102405406882479555878235854939;
where P >= A
q[0] = 4099 - 1 = 2 * 3 * 683;
q[1] = 4111 - 1 = 2 * 3 * 5 * 137;
q[2] = 4127 - 1 = 2 * 2063;
q[3] = 4129 - 1 = (2**5) * 3 * 43;
q[4] = 4133 - 1 = (2**2) * 1033;
q[5] = 4139 - 1 = 2 * 2069;
q[6] = 4153 - 1 = (2**3) * 3 * 173;
q[7] = 4157 - 1 = (2**2) * 1039;
q[8] = 4211 - 1 = 2 * 5 * 421;
```
